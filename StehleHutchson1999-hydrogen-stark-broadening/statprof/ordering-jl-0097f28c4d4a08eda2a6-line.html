123 function _ord(lt, by, order::Ordering)
124 if order === Forward
125 return Lt((x, y) -> lt(by(x), by(y)))
126 elseif order === Reverse
127 return Lt((x, y) -> lt(by(y), by(x)))
128 else
129 error("Passing both lt= and order= arguments is ambiguous; please pass order=Forward or order=Reverse (or leave default)")
130 end

133 """
134 ord(lt, by, rev::Union{Bool, Nothing}, order::Ordering=Forward)
135
136 Construct an [`Ordering`](@ref) object from the same arguments used by
137 [`sort!`](@ref).
138 Elements are first transformed by the function `by` (which may be
139 [`identity`](@ref)) and are then compared according to either the function `lt`
140 or an existing ordering `order`. `lt` should be [`isless`](@ref) or a function
141 which obeys similar rules. Finally, the resulting order is reversed if
142 `rev=true`.
143
144 Passing an `lt` other than `isless` along with an `order` other than
145 [`Base.Order.Forward`](@ref) or [`Base.Order.Reverse`](@ref) is not permitted,
146 otherwise all options are independent and can be used together in all possible
147 combinations.
148 """
149 ord(lt, by, rev::Nothing, order::Ordering=Forward) = _ord(lt, by, order)
150
151 function ord(lt, by, rev::Bool, order::Ordering=Forward)
152 o = _ord(lt, by, order)
153 return rev ? ReverseOrdering(o) : o
154 end
